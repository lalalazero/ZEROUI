## 完全受控

完全受控示例

```jsx
import React, { useState } from 'react'
import { Tree } from 'zero-ui-react'

const App = () => {
    const treeData = [
        {
            title: 'parent 1',
            key: '1',
            children: [
                {
                    title: 'parent 1-1',
                    key: '1-1',
                    children: [
                        {
                            title: 'leaf',
                            key: '1-1-1',
                        },
                        {
                            title: 'leaf',
                            key: '1-1-2',
                        },
                    ],
                },
                {
                    title: <span style={{ color: 'pink' }}>parent 1-2</span>,
                    key: '1-2',
                    children: [
                        {
                            title: 'sss',
                            key: '1-2-1',
                        },
                    ],
                },
            ],
        },
    ]

    const [selectedKeys, setSelectedKeys] = useState(['1-1-1'])
    const [checkedKeys, setCheckedKeys] = useState(['1-1-1'])
    const [expandKeys, setExpandKeys] = useState(['1', '1-2'])

    const handleCheck = (checkedKeys, checkItem, event) => {
        console.log('handleCheck..', checkedKeys)
        setCheckedKeys(checkedKeys)
    }

    const handleSelect = (selectedKeys, node) => {
        console.log('handleSelect..', selectedKeys)
        setSelectedKeys(selectedKeys)
    }

    const handleExpand = (expandKeys, node) => {
        console.log('handleExpand..', expandKeys)
        setExpandKeys(expandKeys)
    }

    return (
        <div>
            <Tree
                treeData={treeData}
                onCheck={handleCheck}
                onSelect={handleSelect}
                onExpand={handleExpand}
                selectedKeys={selectedKeys}
                checkedKeys={checkedKeys}
                expandKeys={expandKeys}
            />
        </div>
    )
}
export default App
```
