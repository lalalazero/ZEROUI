## 基本用法

一个基本的 Tree，包含可勾选、可展开、可选中、禁用等功能

```jsx
import React, { useState } from 'react'
import { Tree } from 'zero-ui-react'

const App = () => {
    const treeData = [
        {
            title: 'parent 1',
            key: '1',
            children: [
                {
                    title: 'parent 1-1',
                    key: '1-1',
                    disabled: true,
                    children: [
                        {
                            title: 'leaf',
                            key: '1-1-1',
                            disableCheckbox: true,
                        },
                        {
                            title: 'leaf',
                            key: '1-1-2',
                        },
                    ],
                },
                {
                    title: <span style={{ color: 'pink' }}>parent 1-2</span>,
                    key: '1-2',
                    children: [
                        {
                            title: 'sss',
                            key: '1-2-1',
                        },
                    ],
                },
            ],
        },
    ]

    const handleCheck = (checkedKeys, checkItem, event) => {
        console.log('checkedKeys ', checkedKeys)
        console.log('checkItem ', checkItem)
        console.log('event ', event)
    }

    const handleSelect = (selectedKeys, node) => {
        console.log('selectedKeys ', selectedKeys)
        console.log('node ', node)
    }

    return (
        <div>
            <Tree
                treeData={treeData}
                onCheck={handleCheck}
                onSelect={handleSelect}
                defaultExpandKeys={['1', '1-1']}
                defaultCheckedKeys={['1', '1-1', '1-2', '1-1-2']}
                defaultSelectedKeys={['1-1-1']}
            />
        </div>
    )
}
export default App
```
