## 基本用法

一个基本的 Tree 展示

```jsx
import React from 'react'
import { Tree } from 'zero-ui-react'

const App = () => {
    const treeData = [
        {
            title: '1',
            key: '1',
            children: [
                {
                    title: '1-1',
                    key: '1-1',
                },
                {
                    title: '1-2',
                    key: '1-2',
                    children: [
                        {
                            title: '1-2-1',
                            key: '1-2-1',
                        },
                    ],
                },
                {
                    title: '1-3',
                    key: '1-3',
                    children: [
                        {
                            title: '1-3-1',
                            key: '1-3-1',
                        },
                    ],
                },
            ],
        },
        {
            title: '2',
            key: '2',
            children: [
                {
                    title: '2-1',
                    key: '2-1',
                },
                {
                    title: '2-2',
                    key: '2-2',
                    children: [
                        {
                            title: '2-2-1',
                            key: '2-2-1',
                        },
                    ],
                },
                {
                    title: '2-3',
                    key: '2-3',
                },
            ],
        },
    ]

    return (
        <div>
            <Tree treeData={treeData}>单身狗</Tree>
        </div>
    )
}
export default App
```
