## 自定义图标

自定义折叠和节点图标

```jsx
import React, { useState } from 'react'
import { Tree, Icon } from 'zero-ui-react'

const App = () => {
    const treeData = [
        {
            title: 'parent 1',
            key: '1',
            icon: <Icon name="happy" />,
            children: [
                {
                    title: 'parent 1-1',
                    key: '1-1',
                    children: [
                        {
                            title: 'leaf',
                            key: '1-1-1',
                        },
                        {
                            title: 'leaf',
                            key: '1-1-2',
                        },
                    ],
                },
                {
                    title: <span style={{ color: 'pink' }}>parent 1-2</span>,
                    icon: <Icon name="joy" />,
                    key: '1-2',
                    children: [
                        {
                            title: 'sss',
                            key: '1-2-1',
                        },
                    ],
                },
            ],
        },
    ]

    const expandIcon = <Icon name="down" />
    const collapseIcon = <Icon name="right" />

    return (
        <div>
            <Tree
                treeData={treeData}
                checkable={false}
                expandIcon={expandIcon}
                collapseIcon={collapseIcon}
            />
        </div>
    )
}
export default App
```
