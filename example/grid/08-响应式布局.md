## 响应式布局

预设6个响应式尺寸：`xs` `sm` `md` `lg` `xl` `xxl` 

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Row, Col } from 'zero-ui-react'
const GridExample = () =>{
    return (
        <div>
            <Row>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
            </Row>
            <Row>
                <Col md={{ span: 12, offset: 1, pull: 1, push: 1, order: 1 }}
                    lg={{ span: 10, offset: 2, pull: 2, push: 2, order: 2 }}>
                    col-1 md-order-1 lg-order-2
                </Col>
                <Col md={{ span: 12, offset: 2, pull: 2, push: 1, order: 2 }}
                    lg={{ span: 10, offset: 2, pull: 2, push: 2, order: 1 }}>
                    col-2 md-order-2 lg-order-1
                </Col>
            </Row> 
        </div>
    )
}
export default GridExample
```

```css
.zeroUI-row {
    margin: 20px 0;
    color: #fff;
}
.zeroUI-col {
    height: 40px;
    line-height: 40px;
    text-align: center;
    
}
.zeroUI-col:nth-child(even) {
    background-color: #87e8de;
}
.zeroUI-col:nth-child(odd) {
    background-color: #36cfc9;
}
```
