## 响应式布局

预设6个响应式尺寸：`xs` `sm` `md` `lg` `xl` `xxl` 

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Row, Col } from 'zero-ui-react'
const GridExample = () =>{
    return (
        <div>
            <Row>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
                <Col xs={24} sm={12} md={8} lg={6} xl={4} xxl={3}>
                    col
                </Col>
            </Row>
        </div>
    )
}
export default GridExample
```
