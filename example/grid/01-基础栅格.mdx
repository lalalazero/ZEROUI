## 基础栅格

使用单一的一组 `Row` 和 `Col` 组件，创建一个 24 等分的网格系统，所有列`Col`必须放在`Row`内。

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Row, Col } from 'zero-ui-react'

const App = () => {
    return (
        <div className="col-with-bg">
            <Row>
                <Col span={12}>col-span-12</Col>
                <Col span={12}>col-span-12</Col>
            </Row>
            <Row>
                <Col span={8}>col-span-8</Col>
                <Col span={8}>col-span-8</Col>
                <Col span={8}>col-span-8</Col>
            </Row>
            <Row>
                <Col span={6}>col-span-6</Col>
                <Col span={6}>col-span-6</Col>
                <Col span={6}>col-span-6</Col>
                <Col span={6}>col-span-6</Col>
            </Row>
        </div>
    )
}
export default App
```

```css
.zeroUI-row {
    margin: 20px 0;
    color: #fff;
}
.zeroUI-col {
    height: 40px;
    line-height: 40px;
    text-align: center;
}
.zeroUI-col:nth-child(even) {
    background-color: #87e8de;
}
.zeroUI-col:nth-child(odd) {
    background-color: #36cfc9;
}
```
