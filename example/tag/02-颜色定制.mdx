## 颜色定制

通过 color 属性定制标签颜色样式

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Tag } from 'zero-ui-react'

const App = () => {
    return (
        <div>
            <Tag color="#f50">#f50</Tag>
            <Tag color="#2db7f5">#2db7f5</Tag>
            <Tag color="#87d068">#87d068</Tag>
            <Tag color="#108ee9">#108ee9</Tag>
        </div>
    )
}
export default App
```
