## 自定义图标

设置 Steps.Step 的 icon 属性

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Steps, Icon } from 'zero-ui-react'

const { Step } = Steps

const App = () => {
    return (
        <Steps current={2}>
            <Step
                title="第一步"
                desc="第一步需要身份证"
                icon={<Icon name="home" className="step-icon-1" />}
            ></Step>
            <Step
                title="第二步"
                desc="第二步需要社保证明"
                icon={<Icon name="loading" className="step-icon-2" />}
            ></Step>
            <Step
                title="第三步"
                desc="等待审核结果"
                icon={<Icon name="joy" className="step-icon-3" />}
            ></Step>
        </Steps>
    )
}
export default App
```
