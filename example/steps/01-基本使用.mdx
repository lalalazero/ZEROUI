## 基本

简单的步骤条

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Steps } from 'zero-ui-react'

const { Step } = Steps

const App = () => {
    return (
        <Steps current={2}>
            <Step title="第一步" desc="第一步需要身份证"></Step>
            <Step title="第二步" desc="第二步需要社保证明"></Step>
            <Step title="第三步" desc="等待审核结果"></Step>
        </Steps>
    )
}
export default App
```
