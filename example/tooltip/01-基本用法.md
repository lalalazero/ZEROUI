## 基本

最简单也最常见的用法。

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { ToolTip } from 'zero-ui-react'

const ToolTipExample = () => {
    return (
        <div>
            <ToolTip title="prompt text">
                <span>鼠标移入展示文字</span>
            </ToolTip>
        </div>
    )
}

export default ToolTipExample 
```
