## 基本

最简单也最常见的用法。

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Tooltip, Button } from 'zero-ui-react'

const App = () => {
    return (
        <div>
            <Tooltip title="prompt text">
                <Button>Tooltip will show on mouse enter.</Button>
            </Tooltip>
        </div>
    )
}

export default App
```
