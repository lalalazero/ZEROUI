## 带图标

带图标的面包屑

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Breadcrumb, Icon } from 'zero-ui-react'

const App = () => {
    return (
        <div>
            <Breadcrumb>
                <Breadcrumb.Item>
                    <Icon name="home" />
                </Breadcrumb.Item>
                <Breadcrumb.Item>
                    <a href="" style={{ verticalAlign: 'middle' }}>
                        <Icon name="wechat" />
                        <span style={{ marginLeft: 5 }}>应用中心</span>
                    </a>
                </Breadcrumb.Item>
                <Breadcrumb.Item>
                    <a href="">应用列表</a>
                </Breadcrumb.Item>
                <Breadcrumb.Item>zeroUI 组件库</Breadcrumb.Item>
            </Breadcrumb>
        </div>
    )
}
export default App
```
