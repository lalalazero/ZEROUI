## 基本使用

包含标题和内容的典型卡片

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Card } from 'zero-ui-react'

const App = () => {
    return (
        <div>
            <Card title={'默认大小卡片'}>
                <p>卡片内容</p>
                <p>卡片内容</p>
                <p>卡片内容</p>
                <p>卡片内容</p>
            </Card>
            <div style={{ height: 20 }}></div>
            <Card title="小尺寸卡片" size="small">
                <p>卡片内容</p>
                <p>卡片内容</p>
                <p>卡片内容</p>
                <p>卡片内容</p>
            </Card>
        </div>
    )
}
export default App
```
