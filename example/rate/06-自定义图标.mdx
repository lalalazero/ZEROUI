## 自定义图标

可以根据需求自定义图标

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Rate, Icon } from 'zero-ui-react'

const customIcons = [
    <Icon name="like" />,
    <Icon name="happy" />,
    <Icon name="joy" />,
    <Icon name="star" />,
    <Icon name="heart" />,
]

const App = () => {
    return (
        <div>
            <Rate charactor="A" />
            <Rate charactor="好" />
            <Rate charactor={(index) => customIcons[index]} />
        </div>
    )
}
export default App
```
