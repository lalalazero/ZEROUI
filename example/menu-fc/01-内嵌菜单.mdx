## 内嵌菜单

竖直方向上打开

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { MenuFC } from 'zero-ui-react'

const { MenuGroup, MenuItem, SubMenu } = MenuFC

const App = () => {
    const handleOpenChange = (openKeys) => {
        console.log('open change..', openKeys)
    }

    const handleSelect = (selectedKeys) => {
        console.log('handle select..', selectedKeys)
    }
    return (
        <div>
            <MenuFC
                defaultOpenKeys={['1-2', '1-2-1', '1-3']}
                defaultSelectedKeys={['1-2-1']}
                onOpenChange={handleOpenChange}
                onSelect={handleSelect}
            >
                <SubMenu title="Navigation One" key="1-1">
                    <MenuGroup title="Item 1">
                        <MenuItem>Option 1</MenuItem>
                        <MenuItem>Option 2</MenuItem>
                    </MenuGroup>
                    <MenuGroup title="Item 2">
                        <MenuItem>Option 3</MenuItem>
                        <MenuItem>Option 4</MenuItem>
                    </MenuGroup>
                </SubMenu>
                <SubMenu title="Navigation Two" key="1-2">
                    <MenuItem>Option 5</MenuItem>
                    <MenuItem>Option 6</MenuItem>
                    <SubMenu title="Submenu" key="1-2-1">
                        <MenuItem>Option 7</MenuItem>
                        <MenuItem>Option 8</MenuItem>
                    </SubMenu>
                </SubMenu>
                <SubMenu title="Navigation Three" key="1-3">
                    <MenuItem>Option 9</MenuItem>
                    <MenuItem>Option 10</MenuItem>
                    <MenuItem>Option 11</MenuItem>
                    <MenuItem>Option 12</MenuItem>
                </SubMenu>
            </MenuFC>
        </div>
    )
}
export default App
```
