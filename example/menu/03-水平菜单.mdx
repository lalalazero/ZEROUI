## 水平菜单

水平的顶部导航菜单

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Menu } from 'zero-ui-react'

const { MenuGroup, MenuItem, SubMenu } = Menu

const App = () => {
    return (
        <div>
            <Menu mode="horizontal">
                <MenuItem>Navigation One</MenuItem>
                <MenuItem>Navigation Two</MenuItem>
                <SubMenu title="Navigation Three">
                    <MenuGroup title="Item 1">
                        <MenuItem>Option 1</MenuItem>
                        <MenuItem>Option 2</MenuItem>
                    </MenuGroup>
                    <MenuGroup title="Item 2">
                        <MenuItem>Option 3</MenuItem>
                        <MenuItem>Option 4</MenuItem>
                    </MenuGroup>
                </SubMenu>
                <MenuItem title="Navigation Four">
                    <a href="https://github.com/lalalazero/zeroui">
                        Link - ZEROUI
                    </a>
                </MenuItem>
            </Menu>
        </div>
    )
}
export default App
```
