## 整数输入框

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { NumberInput } from 'zero-ui-react'

const App = () => {
    const handleChange = (name, value) => {
        console.log('change..', value)
    }
    return <NumberInput min={1} max={10} onChange={handleChange}></NumberInput>
}
export default App
```
