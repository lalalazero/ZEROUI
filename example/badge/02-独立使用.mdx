## 独立使用

不包裹任何元素即为独立使用，可叠加自定义样式。

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Badge } from 'zero-ui-react'

const App = () => {
    return (
        <div>
            <Badge count={5}></Badge>
            <Badge count={4} showZero={true} className="badge-4"></Badge>
            <Badge count={100} style={{ backgroundColor: 'darkgreen' }}></Badge>
        </div>
    )
}
export default App
```
