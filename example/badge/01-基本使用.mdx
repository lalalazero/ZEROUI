## 基本

简单的用法，当 count=0 时默认不展示徽章，可以通过 showZero 修改为显示。

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { Badge, Icon } from 'zero-ui-react'

const App = () => {
    return (
        <div>
            <Badge count={5}>
                <a href="#" className="badge-example" />
            </Badge>
            <Badge count={0} showZero={true}>
                <a href="#" className="badge-example" />
            </Badge>
            <Badge count={<Icon name="joy" style={{ fill: 'red' }} />}>
                <a href="#" className="badge-example" />
            </Badge>
        </div>
    )
}
export default App
```
