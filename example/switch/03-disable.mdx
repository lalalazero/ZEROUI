## 禁用状态

设置禁用

```jsx
import React, { useState } from 'react'
import { Switch, Button } from 'zero-ui-react'

const App = () => {
    const [checked, setChecked] = useState(false)
    const [disabled, setDisabled] = useState(false)
    const onClick = () => {
        setChecked(!checked)
    }

    const handleChange = (name, newCheck) => {
        setChecked(newCheck)
    }

    const toggleDisable = () => setDisabled(!disabled)
    return (
        <div>
            <Switch
                checked={checked}
                disabled={disabled}
                onChange={handleChange}
            ></Switch>
            <p>
                <Button onClick={toggleDisable}>切换禁用状态</Button>
            </p>
        </div>
    )
}
export default App
```
