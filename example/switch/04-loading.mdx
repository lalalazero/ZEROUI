## 加载中

转圈 loading 状态

```jsx
import React from 'react'
import { Switch } from 'zero-ui-react'

const App = () => {
    return (
        <div>
            <Switch checked={true} loading={true} />
            <Switch checked={false} loading={true} />
            <Switch checked={true} loading={true} size="small" />
            <Switch checked={false} loading={true} size="small" />
        </div>
    )
}
export default App
```
