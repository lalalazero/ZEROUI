## 基本用法

基本的文本输入框。

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { TextInput } from 'zero-ui-react'

const App = () => {
    const [inputValue, setInputValue] = React.useState('')
    const onChange = (name, value) => {
        setInputValue(value)
    }
    return (
        <TextInput
            placeholder={'enter your name'}
            value={inputValue}
            onChange={onChange}
        ></TextInput>
    )
}
export default App
```
