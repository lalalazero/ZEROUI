## 尺寸大小

三种不同尺寸的输入框

```jsx
import React from 'react'
import ReactDOM from 'react-dom'
import { TextInput } from 'zero-ui-react'

const App = () => {
    const [inputValue, setInputValue] = React.useState('')
    const onChange = (name, value) => {
        setInputValue(value)
    }
    return (
        <>
            <div style={{ marginBottom: 16 }}>
                <TextInput
                    placeholder="large"
                    size="large"
                    value={inputValue}
                    onChange={onChange}
                ></TextInput>
            </div>

            <div style={{ marginBottom: 16 }}>
                <TextInput
                    placeholder="default"
                    value={inputValue}
                    onChange={onChange}
                ></TextInput>
            </div>
            <div style={{ marginBottom: 16 }}>
                <TextInput
                    placeholder="small"
                    size="small"
                    value={inputValue}
                    onChange={onChange}
                ></TextInput>
            </div>
        </>
    )
}
export default App
```
