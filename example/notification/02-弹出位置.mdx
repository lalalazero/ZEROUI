## 弹出位置

指定弹出位置

```jsx
import React from 'react'
import { notification, Button } from 'zero-ui-react'

const App = () => {
    const defaultConfig = {
        title: '标题',
        body: '通知内容',
        autoClose: false,
    }
    const open = (placement) =>
        notification.open({ ...defaultConfig, placement })
    return (
        <div className="btn-wrapper">
            <Button onClick={() => open('topRight')}>右上角弹出</Button>
            <Button onClick={() => open('topLeft')}>左上角弹出</Button>
            <Button onClick={() => open('bottomRight')}>右下角弹出</Button>
            <Button onClick={() => open('bottomLeft')}>左下角弹出</Button>
            <Button onClick={() => open('topCenter')}>中间顶部弹出</Button>
            <Button onClick={() => open('bottomCenter')}>中间底部弹出</Button>
        </div>
    )
}
export default App
```

```css
.btn-wrapper > .zeroUI-button {
    margin: 5px;
}
```
