## 弹出位置

指定弹出位置

```jsx
import React from 'react'
import { notification, Button } from 'zero-ui-react'

const App = () => {
    const defaultConfig = {
        title: <h4>标题</h4>,
        body: '通知内容',
    }
    const open = (placement) =>
        notification.open({ ...defaultConfig, placement })
    return (
        <div className="btn-wrapper">
            <Button onClick={() => open('topRight')}>topRight</Button>
            <Button onClick={() => open('topLeft')}>topLeft</Button>
            <Button onClick={() => open('bottomRight')}>bottomRight</Button>
            <Button onClick={() => open('bottomLeft')}>bottomLeft</Button>
            <Button onClick={() => open('topCenter')}>topCenter</Button>
            <Button onClick={() => open('bottomCenter')}>bottomCenter</Button>
        </div>
    )
}
export default App
```

```css
.btn-wrapper > .zeroUI-button {
    margin: 5px;
}
```
