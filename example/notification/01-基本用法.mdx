## 基本用法

简单使用，打开后 5s 自动关闭

```jsx
import React from 'react'
import { notification, Button } from 'zero-ui-react'

const App = () => {
    const onClick = () => {
        notification.open({
            title: '标题',
            body: '通知内容',
        })
    }
    return <Button onClick={onClick}>打开通知</Button>
}
export default App
```
