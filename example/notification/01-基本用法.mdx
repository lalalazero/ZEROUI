## 基本用法

简单使用，打开后 5s 自动关闭

```jsx
import React from 'react'
import { Button, notification } from 'zero-ui-react'

const App = () => {
    const onClick = () => {
        notification.open({
            title: <h4>标题</h4>,
            body: '通知内容',
        })
    }
    return (
        <div>
            <Button onClick={onClick}>打开通知</Button>
        </div>
    )
}
export default App
```
