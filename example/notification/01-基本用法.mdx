## 基本用法

简单使用，打开后 5s 自动关闭

```jsx
import React from 'react'
import { Button, notification } from 'zero-ui-react'

const App = () => {
    const onClick2 = () => {
        notification.open({
            title: '标题222',
            body: '通知内容222',
        })
    }
    return (
        <div>
            <Button onClick={onClick2} style={{ marginLeft: 10 }}>
                打开通知-2
            </Button>
        </div>
    )
}
export default App
```
